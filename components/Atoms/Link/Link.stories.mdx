import { Meta, Story, Canvas } from "@storybook/addon-docs";

import Link from "./Link.vue";

export const argTypes = {
  linkType: {
    control: "radio",
    options: ["nuxt-link", "normal-link"],
    defaultValue: "nuxt-link",
  },
};

<!--👇 The title prop determines where your story goes in the story list -->

<Meta title="Atoms/Link" component={Link} argTypes={argTypes} />

export const Template = (args, { argTypes }) => ({
  props: Object.keys(argTypes),
  components: { Link },
  template: `<Link :href="href" :target="target" :linkType="linkType">{{ content }}</Link>`,
});
